DRF_Class_Based_Views_Hierarchy:
  - APIView:
      description: "Base low-level view (manual methods, max control)"
      children:
        - GenericAPIView:
            description: "Adds queryset, serializer support, filtering, pagination"
            children:
              - Mixins:
                  description: "Reusable behavior pieces (used WITH GenericAPIView)"
                  children:
                    - CreateModelMixin: "adds .create() for POST"
                    - RetrieveModelMixin: "adds .retrieve() for GET single"
                    - UpdateModelMixin: "adds .update(), .partial_update()"
                    - DestroyModelMixin: "adds .destroy() for DELETE"
                    - ListModelMixin: "adds .list() for GET list"
              - Concrete_Generic_Views:
                  description: "Combine GenericAPIView + mixins for common CRUD"
                  children:
                    - CreateAPIView: "POST only"
                    - ListAPIView: "GET list only"
                    - RetrieveAPIView: "GET single only"
                    - UpdateAPIView: "PUT/PATCH only"
                    - DestroyAPIView: "DELETE only"
                    - ListCreateAPIView: "GET list + POST create"
                    - RetrieveUpdateDestroyAPIView: "GET/PUT/PATCH/DELETE single item"
        - ViewSets:
        
            description: "Combine mixins into a single class for full resource CRUD; used with routers"
            children:
              - ViewSet:
                  description: "Base class; actions must be defined manually"
              - GenericViewSet:
                  description: "Combines GenericAPIView with ViewSet base; supports queryset and serializers"
              - ReadOnlyModelViewSet:
                  description: "Provides .list() and .retrieve() actions (read-only endpoints)"
              - ModelViewSet:
                  description: "Full CRUD viewset combining all mixins (list, create, retrieve, update, destroy)"
